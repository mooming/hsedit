5. Input Handling

Basic getch()

// Blocking: waits for a key press
int key = getch();

// Non-blocking: returns ERR (-1) if no key pressed
nodelay(stdscr, TRUE);
int key = getch();   // Returns ERR if nothing pressed

Modifiers (Ctrl, Shift, Alt)

// Ctrl combinations are single bytes:
int key = getch();
if (key == 3)  // Ctrl+C (ASCII 3)
if (key == 21) // Ctrl+U
if (key == 19) // Ctrl+S  (common: save)
if (key == 1)  // Ctrl+A  (common: command mode)

// Alt + character: ncurses sends ESC first, then the character
if (key == 27) {  // ESC pressed — check if followed by a char
    nodelay(stdscr, TRUE);
    int next = getch();
    if (next != ERR) {
        // Alt+key: ESC + character = 27 + next
        handle_alt_key(next);
    }
}

Input Buffers

// Clear the input buffer (discard pending keystrokes)
flushinp();

// Set a timeout — getch returns ERR after ms milliseconds
timeout(100);    // 100ms timeout
halfdelay(5);   // 0.5 second timeout (halfdelay mode)

6. Output & Drawing

Writing Text

// Print at current cursor position
printw("Hello at cursor");

// Print at specific (y, x) position
mvprintw(5, 10, "Hello at (5,10)");

// For a specific window
wprintw(win, "In sub-window");
mvwprintw(win, 2, 3, "Specific spot");

// Individual character
addch('A');
mvaddch(3, 4, 'X');

Attributes (Bold, Underline, etc.)

// Set attributes
attron(A_BOLD);        // Bold text
printw("This is bold");
attroff(A_BOLD);       // Turn off bold

// Multiple attributes combined
attron(A_BOLD | A_UNDERLINE);

// Common attributes:
// A_BOLD, A_DIM, A_UNDERLINE, A_REVERSE, A_BLINK
// A_PROTECT, A_INVIS, A_STANDOUT

Clearing

clear();         // Clear entire screen
erase();         // Same as clear() but for stdscr
wclear(win);    // Clear a specific window
clearok(stdscr, TRUE);  // Force clear on next refresh

Scrolling

scroll(stdscr);     // Scroll window up by one line
scrl(-3);          // Scroll down by 3 lines
scrl(2);           // Scroll up by 2 lines

Lines and Boxes

// Draw a box around a window
box(win, 0, 0);     // Default border characters
box(win, '|', '-');   // Custom border chars
