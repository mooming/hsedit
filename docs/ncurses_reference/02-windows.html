Window Management

Home› Windows

newwin(int nlines, int ncols, int begin_y, int begin_x)

Creates a new window.

WINDOW* subwin = newwin(5, 40, 5, 10);
if (subwin)
{
    wprintw(subwin, "Sub-window content");
    wrefresh(subwin);
}

delwin(WINDOW* win)

Deletes a window and frees its memory.

WINDOW* win = newwin(10, 20, 0, 0);
// ... use window ...
delwin(win);

mvwin(WINDOW* win, int y, int x)

Moves a window to a new position.

WINDOW* win = newwin(5, 20, 0, 0);
mvprintw(0, 0, "Before move");
mvwin(win, 10, 10);
mvprintw(0, 0, "After move");
refresh();

wresize(WINDOW* win, int nlines, int ncols)

Changes the size of an existing window.

WINDOW* win = newwin(5, 20, 0, 0);
wresize(win, 10, 40);
mvprintw(0, 0, "Resized window");
wrefresh(win);

subwin(WINDOW* orig, int nlines, int ncols, int begin_y, int begin_x)

Creates a sub-window that shares memory with the original window.

WINDOW* main_win = initscr();
WINDOW* sub = subwin(main_win, 5, 20, 5, 5);
wattron(sub, A_REVERSE);
wprintw(sub, "Sub-window");
wrefresh(sub);

derwin(WINDOW* orig, int nlines, int ncols, int begin_y, int begin_x)

Like subwin() but does not share memory.

mvderwin(WINDOW* win, int pary, int parx)

Moves a window derived from another window.

dupwin(WINDOW* win)

Creates an exact copy of a window.

window_ok(WINDOW* win)

Returns true if win is a valid window pointer.

touchwin(WINDOW* win)

Marks all lines in a window as needing refresh.

WINDOW* win = newwin(10, 40, 0, 0);
touchwin(win);
wrefresh(win);

touchline(WINDOW* win, int start, int count)

Marks a range of lines in a window as needing refresh.

untouchwin(WINDOW* win)

Marks all lines in a window as not needing refresh.

redrawwin(WINDOW* win)

Forces refresh of the entire window.

redrawline(WINDOW* win, int start, int count)

Forces refresh of a range of lines.

wclear(WINDOW* win)

Clears a window (fills with spaces).

werase(WINDOW* win)

Like wclear() but marks window as needing refresh.

waddch(WINDOW* win, chtype ch)

Adds a character to a window.

WINDOW* win = newwin(5, 20, 0, 0);
waddch(win, 'A');
waddch(win, 'B');
wrefresh(win);

wmove(WINDOW* win, int y, int x)

Moves the cursor within a window.

WINDOW* win = newwin(10, 40, 0, 0);
wmove(win, 5, 10);
wprintw(win, "Cursor at 5,10");
wrefresh(win);

wrefresh(WINDOW* win)

Updates the physical screen to match a window.

WINDOW* win = newwin(10, 40, 0, 0);
wprintw(win, "Content");
wrefresh(win);

wnoutrefresh(WINDOW* win)

Like wrefresh() but does not call doupdate().

WINDOW* win1 = newwin(5, 20, 0, 0);
WINDOW* win2 = newwin(5, 20, 0, 25);
wprintw(win1, "Window 1");
wprintw(win2, "Window 2");
wnoutrefresh(win1);
wnoutrefresh(win2);
doupdate();

wborder(WINDOW* win, ...)

Draws a border around a window.

WINDOW* win = newwin(10, 40, 5, 5);
wborder(win, '|', '|', '-', '-', '+', '+', '+', '+');
wrefresh(win);

wattron(WINDOW* win, attr_t attr)

TURNS on attributes (bold, reverse, etc.).

WINDOW* win = newwin(5, 20, 0, 0);
wattron(win, A_BOLD);
wprintw(win, "Bold text");
wattroff(win, A_BOLD);
wrefresh(win);

wattroff(WINDOW* win, attr_t attr)

TURNS off attributes.
