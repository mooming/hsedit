printw(const char* fmt, ...)

Formatted print to stdscr (like printf).

int x = 42;
printw("The answer is %d", x);
refresh();

wprintw(WINDOW* win, const char* fmt, ...)

Formatted print to a window.

WINDOW* win = newwin(5, 20, 0, 0);
wprintw(win, "Value: %d", 42);
wrefresh(win);

mvprintw(int y, int x, const char* fmt, ...)

Move to position and print to stdscr.

mvprintw(10, 20, "Hello at 10,20");
refresh();

mvwprintw(WINDOW* win, int y, int x, const char* fmt, ...)

Move to position in window and print.

addch(chtype ch)

Add a character to stdscr.

addch('A');
addch('B' | A_BOLD);  // Bold B
refresh();

waddch(WINDOW* win, chtype ch)

Add a character to a window.

mvaddch(int y, int x, chtype ch)

Move and add a character to stdscr.

mvwaddch(WINDOW* win, int y, int x, chtype ch)

Move and add a character to a window.

addstr(const char* str)

Add a string to stdscr.

addstr("Hello, World!");
refresh();

waddstr(WINDOW* win, const char* str)

Add a string to a window.

mvaddstr(int y, int x, const char* str)

Move and add a string to stdscr.

mvwaddstr(WINDOW* win, int y, int x, const char* str)

Move and add a string to a window.

slk_init(int format)

Initialize soft keys (3270 terminals).

slk_init(SLK_STDMODE);
slk_set(1, 1, "Help", 1);
slk_set(2, 1, "Exit", 1);
slk_refresh();

slk_set(int labnum, int align, const char* label, int refresh)

Set a soft key label.

slk_refresh()

Redraw soft keys.

slk_clear()

Clear all soft keys.

slk_restore()

Restore soft keys.
